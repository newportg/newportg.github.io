Example: “Order Processor” Microservice Requirement

Section AI-Ready Specification
Context You are a Senior Azure Developer. I need an Azure Function (v4, .NET 8 Isolated) that acts as an “Order Processor” microservice. It is part of a serverless e-commerce architecture and must be stateless and defensive.
Action Create an HTTP-triggered function (POST only) that:
1. Validates a JSON body against an Order model (fields: OrderId, CustomerEmail, TotalAmount).
2. Persists the data to Azure Table Storage using Output Bindings to avoid SDK boilerplate.
3. Logs successful executions to Application Insights with structured logging.
Result Code must include:
• Proper try-catch blocks with specific error codes (e.g., 400 for validation, 500 for storage errors).
• Asynchronous/await patterns for all I/O operations.
• Managed Identity support (no hardcoded connection strings).
• Unit test stubs using xUnit and Moq.
Example Input JSON {“OrderId”: “123”, “CustomerEmail”: “user@test.com”, “TotalAmount”: 99.99}
Success Response 201 Created with a location header.
Failure Example If TotalAmount is negative, return 400 Bad Request with message “Total cannot be less than zero”.

Azure Function : OrderProcessor

using System.Net;
using Microsoft.Azure.Functions.Worker;
using Microsoft.Azure.Functions.Worker.Http;
using Microsoft.Extensions.Logging;

public class OrderProcessor
{
    private readonly ILogger _logger;

    public OrderProcessor(ILoggerFactory loggerFactory)
    {
        _logger = loggerFactory.CreateLogger<OrderProcessor>();
    }

    [Function("ProcessOrder")]
    public async Task<OrderResponse> Run(
        [HttpTrigger(AuthorizationLevel.Function, "post")] HttpRequestData req)
    {
        _logger.LogInformation("Processing new order request.");

        var order = await req.ReadFromJsonAsync<OrderEntity>();

        // Validation Logic
        if (order == null || string.IsNullOrEmpty(order.OrderId) || order.TotalAmount < 0)
        {
            _logger.LogWarning("Invalid order received: {OrderId}", order?.OrderId ?? "NULL");
            return new OrderResponse 
            { 
                HttpResponse = req.CreateResponse(HttpStatusCode.BadRequest) 
            };
        }

        // Prepare data for Table Storage
        order.PartitionKey = "Orders";
        order.RowKey = order.OrderId;

        var response = req.CreateResponse(HttpStatusCode.Created);
        response.Headers.Add("Location", $"/api/orders/{order.OrderId}");

        return new OrderResponse
        {
            NewOrder = order,
            HttpResponse = response
        };
    }
}

public class OrderResponse
{
    // Write to "OrdersTable" using the connection "TableStorageConnection"
    [TableOutput("OrdersTable", Connection = "TableStorageConnection")]
    public OrderEntity NewOrder { get; set; }

    public HttpResponseData HttpResponse { get; set; }
}

Unit Test Stub: OrderProcessorTests

using Microsoft.Azure.Functions.Worker;
using Microsoft.Azure.Functions.Worker.Http;
using Microsoft.Extensions.Logging;
using Moq;
using System.Net;
using System.Text;
using Xunit;

public class OrderProcessorTests
{
    private readonly Mock<ILoggerFactory> _mockLoggerFactory;
    private readonly Mock<ILogger<OrderProcessor>> _mockLogger;
    private readonly OrderProcessor _sut; // System Under Test

    public OrderProcessorTests()
    {
        _mockLoggerFactory = new Mock<ILoggerFactory>();
        _mockLogger = new Mock<ILogger<OrderProcessor>>();
        
        _mockLoggerFactory
            .Setup(x => x.CreateLogger(It.IsAny<string>()))
            .Returns(_mockLogger.Object);

        _sut = new OrderProcessor(_mockLoggerFactory.Object);
    }

    [Fact]
    public async Task Run_ShouldReturnBadRequest_WhenTotalAmountIsNegative()
    {
        // Arrange
        var context = new Mock<FunctionContext>();
        var request = MockHelpers.CreateMockRequest("{\"OrderId\": \"123\", \"TotalAmount\": -10.0}");

        // Act
        var result = await _sut.Run(request.Object);

        // Assert
        Assert.Equal(HttpStatusCode.BadRequest, result.HttpResponse.StatusCode);
        _mockLogger.VerifyLog(LogLevel.Warning, "Invalid order received");
    }

    [Fact]
    public async Task Run_ShouldReturnCreated_WhenOrderIsValid()
    {
        // Arrange
        var request = MockHelpers.CreateMockRequest("{\"OrderId\": \"ORD-99\", \"CustomerEmail\": \"test@test.com\", \"TotalAmount\": 50.0}");

        // Act
        var result = await _sut.Run(request.Object);

        // Assert
        Assert.Equal(HttpStatusCode.Created, result.HttpResponse.StatusCode);
        Assert.NotNull(result.NewOrder);
        Assert.Equal("ORD-99", result.NewOrder.RowKey);
    }
}

MockHelpers.cs

public static class MockHelpers
{
    public static Mock<HttpRequestData> CreateMockRequest(string jsonBody)
    {
        var context = new Mock<FunctionContext>();
        var request = new Mock<HttpRequestData>(context.Object);
        
        var stream = new MemoryStream(Encoding.UTF8.GetBytes(jsonBody));
        request.Setup(r => r.Body).Returns(stream);
        
        // Mock CreateResponse to return a real HttpResponseData implementation
        request.Setup(r => r.CreateResponse()).Returns(() => 
        {
            var response = new Mock<HttpResponseData>(context.Object);
            response.SetupProperty(r => r.StatusCode);
            response.Setup(r => r.Headers).Returns(new HttpHeadersCollection());
            return response.Object;
        });

        return request;
    }
}
